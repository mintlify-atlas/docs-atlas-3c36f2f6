## Overview

This guide will walk you through creating a fully functional Discord bot using Magicyan. You'll learn how to:

- Set up a Discord bot with custom intents
- Create commands with slash command handlers
- Use Magicyan helpers for channels, embeds, and messages
- Handle interactions with simplified APIs

<Note>
Make sure you've completed the [installation](/installation) before continuing.
</Note>

## Prerequisites

Before you begin, you'll need:

1. A Discord Bot Token from the [Discord Developer Portal](https://discord.com/developers/applications)
2. Node.js 18+ installed
3. Magicyan packages installed

## Project Setup

<Steps>
  <Step title="Create your project structure">
    Create a new directory and initialize your project:

    ```bash
    mkdir my-discord-bot
    cd my-discord-bot
    npm init -y
    ```

    Set up the file structure:

    ```bash
    mkdir src
    touch src/index.ts
    touch .env
    ```
  </Step>

  <Step title="Install dependencies">
    Install the required packages:

    <CodeGroup>
    ```bash npm
    npm install discord.js @magicyan/discord @magicyan/discord-events
    npm install -D typescript tsx @types/node
    ```

    ```bash yarn
    yarn add discord.js @magicyan/discord @magicyan/discord-events
    yarn add -D typescript tsx @types/node
    ```

    ```bash pnpm
    pnpm add discord.js @magicyan/discord @magicyan/discord-events
    pnpm add -D typescript tsx @types/node
    ```
    </CodeGroup>
  </Step>

  <Step title="Configure TypeScript">
    Create a `tsconfig.json` file:

    ```json tsconfig.json
    {
      "compilerOptions": {
        "target": "ES2022",
        "module": "ESNext",
        "moduleResolution": "node",
        "esModuleInterop": true,
        "strict": true,
        "skipLibCheck": true,
        "resolveJsonModule": true,
        "outDir": "./dist",
        "rootDir": "./src"
      },
      "include": ["src/**/*"]
    }
    ```
  </Step>

  <Step title="Add environment variables">
    Create a `.env` file with your bot token:

    ```bash .env
    BOT_TOKEN=your_discord_bot_token_here
    ```

    <Warning>
    Never commit your `.env` file to version control. Add it to `.gitignore`.
    </Warning>
  </Step>

  <Step title="Update package.json">
    Add scripts and set module type:

    ```json package.json
    {
      "name": "my-discord-bot",
      "version": "1.0.0",
      "type": "module",
      "scripts": {
        "dev": "tsx src/index.ts",
        "build": "tsc",
        "start": "node dist/index.js"
      }
    }
    ```
  </Step>
</Steps>

## Create Your Bot

<Steps>
  <Step title="Initialize the Discord client">
    Create your bot with Magicyan's custom intents:

    ```typescript src/index.ts
    import { Client } from "discord.js";
    import { CustomItents } from "@magicyan/discord";
    import { initDiscordEvents } from "@magicyan/discord-events";

    // Create client with all intents
    const client = new Client({
      intents: CustomItents.All
    });

    // Initialize extended events
    initDiscordEvents(client);
    ```

    <Tip>
    `CustomItents.All` includes all available intents. For production, use specific intents like `CustomItents.Guild` or `CustomItents.Messages` to reduce memory usage.
    </Tip>
  </Step>

  <Step title="Handle the ready event">
    Add a handler for when your bot comes online:

    ```typescript src/index.ts
    client.on("ready", (client) => {
      console.log(`✅ ${client.user.username} is online!`);
      
      // Register a test slash command
      client.application.commands.set([
        {
          name: "hello",
          description: "Say hello to the bot"
        },
        {
          name: "info",
          description: "Get bot information"
        }
      ]);
    });
    ```
  </Step>

  <Step title="Create slash command handlers">
    Handle slash commands using Magicyan's helpers:

    ```typescript src/index.ts
    import { createEmbed, findChannel } from "@magicyan/discord";
    import { ChannelType } from "discord.js";

    client.on("interactionCreate", async (interaction) => {
      // Handle slash commands
      if (!interaction.isChatInputCommand()) return;

      switch (interaction.commandName) {
        case "hello": {
          // Create a rich embed using Magicyan
          const embed = createEmbed({
            title: "👋 Hello!",
            description: [
              "Welcome to Magicyan!",
              "I'm a Discord bot built with the Magicyan SDK.",
              "",
              "Try the `/info` command to learn more!"
            ],
            color: "#5865F2",
            timestamp: true,
            footer: "Powered by Magicyan"
          });

          await interaction.reply({ embeds: [embed] });
          break;
        }

        case "info": {
          const guild = interaction.guild;
          if (!guild) return;

          // Use Magicyan's channel finder
          const textChannels = guild.channels.cache.filter(
            c => c.type === ChannelType.GuildText
          );

          const embed = createEmbed({
            title: "📊 Bot Information",
            color: "#00FF00",
            fields: [
              {
                name: "Server Name",
                value: guild.name,
                inline: true
              },
              {
                name: "Member Count",
                value: `${guild.memberCount} members`,
                inline: true
              },
              {
                name: "Text Channels",
                value: `${textChannels.size} channels`,
                inline: true
              },
              {
                name: "Bot Latency",
                value: `${client.ws.ping}ms`,
                inline: true
              }
            ],
            timestamp: true,
            footer: { text: `Requested by ${interaction.user.tag}`, iconURL: interaction.user.displayAvatarURL() }
          });

          await interaction.reply({ embeds: [embed] });
          break;
        }
      }
    });
    ```
  </Step>

  <Step title="Use extended events">
    Listen to Magicyan's extended Discord events:

    ```typescript src/index.ts
    // Handle member timeout events
    client.on("guildMemberTimeoutAdd", (member, timeout) => {
      console.log(`⏰ ${member.user.tag} was timed out until ${timeout.timeout}`);
    });

    client.on("guildMemberTimeoutRemove", (member) => {
      console.log(`✅ Timeout removed for ${member.user.tag}`);
    });

    // Handle voice events
    client.on("guildMemberConnect", (member, channel) => {
      console.log(`🔊 ${member.user.tag} joined ${channel.name}`);
    });

    client.on("guildMemberDisconnect", (member, channel) => {
      console.log(`🔇 ${member.user.tag} left ${channel.name}`);
    });

    // Handle kick/ban events
    client.on("userKick", (user, guild, executor) => {
      console.log(`👢 ${user.tag} was kicked from ${guild.name}`);
    });
    ```

    <Info>
    These extended events are provided by `@magicyan/discord-events` and automatically detect moderation actions from audit logs.
    </Info>
  </Step>

  <Step title="Login to Discord">
    Finally, log in with your bot token:

    ```typescript src/index.ts
    // Login to Discord
    client.login(process.env.BOT_TOKEN);
    ```
  </Step>
</Steps>

## Complete Example

Here's the complete bot code:

```typescript src/index.ts
import { Client, ChannelType } from "discord.js";
import { CustomItents, createEmbed, findChannel } from "@magicyan/discord";
import { initDiscordEvents } from "@magicyan/discord-events";

// Create client with all intents
const client = new Client({
  intents: CustomItents.All
});

// Initialize extended events
initDiscordEvents(client);

// Bot ready event
client.on("ready", (client) => {
  console.log(`✅ ${client.user.username} is online!`);
  
  // Register slash commands
  client.application.commands.set([
    {
      name: "hello",
      description: "Say hello to the bot"
    },
    {
      name: "info",
      description: "Get bot information"
    }
  ]);
});

// Slash command handler
client.on("interactionCreate", async (interaction) => {
  if (!interaction.isChatInputCommand()) return;

  switch (interaction.commandName) {
    case "hello": {
      const embed = createEmbed({
        title: "👋 Hello!",
        description: [
          "Welcome to Magicyan!",
          "I'm a Discord bot built with the Magicyan SDK.",
          "",
          "Try the `/info` command to learn more!"
        ],
        color: "#5865F2",
        timestamp: true,
        footer: "Powered by Magicyan"
      });

      await interaction.reply({ embeds: [embed] });
      break;
    }

    case "info": {
      const guild = interaction.guild;
      if (!guild) return;

      const textChannels = guild.channels.cache.filter(
        c => c.type === ChannelType.GuildText
      );

      const embed = createEmbed({
        title: "📊 Bot Information",
        color: "#00FF00",
        fields: [
          {
            name: "Server Name",
            value: guild.name,
            inline: true
          },
          {
            name: "Member Count",
            value: `${guild.memberCount} members`,
            inline: true
          },
          {
            name: "Text Channels",
            value: `${textChannels.size} channels`,
            inline: true
          },
          {
            name: "Bot Latency",
            value: `${client.ws.ping}ms`,
            inline: true
          }
        ],
        timestamp: true,
        footer: { text: `Requested by ${interaction.user.tag}`, iconURL: interaction.user.displayAvatarURL() }
      });

      await interaction.reply({ embeds: [embed] });
      break;
    }
  }
});

// Extended events
client.on("guildMemberTimeoutAdd", (member, timeout) => {
  console.log(`⏰ ${member.user.tag} was timed out`);
});

client.on("guildMemberConnect", (member, channel) => {
  console.log(`🔊 ${member.user.tag} joined ${channel.name}`);
});

// Login
client.login(process.env.BOT_TOKEN);
```

## Tips for Production

<Tip>
**Use specific intents** - Instead of `CustomItents.All`, use only what you need:

```typescript
const client = new Client({
  intents: [
    ...CustomItents.Guild,
    ...CustomItents.Messages
  ]
});
```
</Tip>

<Tip>
**Handle errors gracefully** - Add error handlers:

```typescript
client.on("error", (error) => {
  console.error("Client error:", error);
});

process.on("unhandledRejection", (error) => {
  console.error("Unhandled rejection:", error);
});
```
</Tip>

<Tip>
**Use environment variables** - Never hardcode sensitive data like tokens.
</Tip>
